namespace = ATTILA

# The Creation of Normandy
# in Brittany
narrative_event = {
	id = ATTILA.001
	desc = EVTDESC_ATTILA_001
	title = EVTTITLE_ATTILA_001
	picture = GFX_evt_book_carolingian_script

	major = yes
	
	trigger = {
		has_landed_title = d_normandy
		OR = {
			culture = norse
			culture = norman
			religion = norse_pagan
		}
		NOT = { has_global_flag = normandy_established }
	}

	major_trigger = {
		capital_scope = {
			NOT = {
				distance = {
					where = 106 # Nantes
					value = 800	# Same part of the world
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		set_global_flag = normandy_established
	}
	
	option = {
		name = EVTOPTA_ATTILA_001
		trigger = {
			has_landed_title = d_normandy
		}
		d_normandy = {
			set_name = ""
			adjective = ""
			reset_coa = THIS
		}
	}
	option = {
		name = EVTOPTB_ATTILA_001
		trigger = {
			NOT = { has_landed_title = d_normandy }
			culture = norse
		}
	}
	option = {
		name = EVTOPTC_ATTILA_001
		trigger = {
			NOT = { has_landed_title = d_normandy }
			NOT = { culture = norse }
		}
	}
}
